152 public void visualizzaBraniPlaylist() {
153 //Creazione della tabella
154 DefaultTableModel model = (DefaultTableModel) this.jTable2.getModel();
155 // Arraylist di tipo canzone, conterrà i brani della playlist
156 ArrayList<Canzoni> brani = new ArrayList<Canzoni>();
157
158 try {
159 // Chiamo il metodo getBrani sulla playlist, per ottenere i brani
160 brani = is.getCanzoniFromPlaylist(pl.getIdPlaylist());
161 } catch (CanzoneInesistente ex) {
162 JOptionPane.showMessageDialog(null, "Canzone inesistente", "Errore", JOptionPane.WARNING_MESSAGE);
163 } catch (RemoteException |MyServerException ex) {
164 JOptionPane.showMessageDialog(null, "Errore", "Errore", JOptionPane.WARNING_MESSAGE);
165 }
166
167 // Creo un Vector di tipo String, che serve per inserire i brani nella tabella
168 Vector<String> vector = new Vector<>();
169 model.setRowCount(0); // Serve per resettare la tabella
170
171 // For che inserisce tutti i brani nella tabella
172 for (int i = 0; i < brani.size(); i++) {
173 // Metodo che trasforma l'ArrayList in un Vector
174 vector = transform(i, brani);
175 if (vector.get(0).equals("Errore #B0001.indexNonValido")) {
176 JOptionPane.showMessageDialog(null, vector.get(0), "Errore", JOptionPane.ERROR_MESSAGE);
177 break;
178 } else {
179 // Aggiungo la riga
180 model.addRow(vector);
181 }
182
183 }
184
185 }

193 private Vector<String> transform(int index, ArrayList<Canzoni> brani) {
194 Vector<String> vector = new Vector<>();
195 if (index < brani.size()) {
196 // Aggiungo le varie informazioni del brano
197 vector.add(brani.get(index).getTitolo());
198 vector.add(brani.get(index).getAutore());
199 vector.add(String.valueOf(brani.get(index).getAnno()));
200 return vector;
201 } else {
202 vector.add("Errore #B0001.indexNonValido"); // Errore
203 return vector;
204 }
