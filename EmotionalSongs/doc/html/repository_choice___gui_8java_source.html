159 private Vector<String> transform(int index, ArrayList<Canzoni> brani) {
160 Vector<String> vector = new Vector<>();
161 if (index < brani.size()) {
162 // Aggiungo le varie informazioni del brano
163 vector.add(brani.get(index).getTitolo());
164 vector.add(brani.get(index).getAutore());
165 vector.add(String.valueOf(brani.get(index).getAnno()));
166 return vector;
167 } else {
168 vector.add("Errore #B0001.indexNonValido"); // Errore
169 return vector;
170 }
171 }
