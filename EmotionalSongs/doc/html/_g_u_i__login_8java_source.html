139 //assegno alla variabile user il contenuto della textbox dove l'utente inserirà il suo UserId/codice fiscale
140 String codFisc = txtUserId.getText();
141 //assegno alla variabile password il contenuto della textbox dove l'utente inserirà la sua password
142 String password = txtPassword.getText();
143 UtentiRegistrati utente;
144 boolean check=false;
145
146 try{
147 //metodo che effettua il login
148 utente =is.login(codFisc, password);
149 check = true;
150 }catch (UtenteInesistente ex) {
151 JOptionPane.showMessageDialog(null, "Utente non registrato", "Errore", JOptionPane.ERROR_MESSAGE);
152 } catch (PasswordErrata ex) {
153 JOptionPane.showMessageDialog(null, "Password errata", "Errore", JOptionPane.ERROR_MESSAGE);
154 } catch (MyServerException | RemoteException ex) {
155 Logger.getLogger(GUI_login.class.getName()).log(Level.SEVERE, null, ex);
156 }
157
158 //dopo aver verificato che il login sia andato a buon termine, invoco il JPanel successivo
159 if(check){
160 JOptionPane.showMessageDialog(null, "Benvenuto: " + codFisc, "Info", JOptionPane.INFORMATION_MESSAGE);
161 this.dispose();
162 final JDialog frame = new JDialog(this, "Gui_Playlist", true);
163 //aggiunta dell'istanza del nuovo JPanel Gui_Playlist al JDialog.
164 frame.getContentPane().add(new Gui_Playlist(codFisc,previousframe,is));
165 //packing del nuovo JPanel
166 frame.pack();
167 //visualizzazione del nuovo JPanel appena realizzato
168 frame.setVisible(true);
169 }
