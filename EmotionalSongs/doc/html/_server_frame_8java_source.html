168 private boolean checkPsw() {
169 try {
170 //istanzio un buffer per la lettura di stringhe dal file degli accessi
171 BufferedReader br = new BufferedReader(new FileReader("accessodb.csv"));
172
173 String text;
174 //lettura di ciascuna riga
175 while ((text = br.readLine()) != null) {
176 //split sul campo ;
177 String[] splittedline = text.split(";");
178 //se ottengo 2 campi, controllo se coincidono con le credenziali
179 if (splittedline.length == 2) {
180 if (splittedline[0].equals(txtUsername.getText()) && splittedline[1].equals(String.valueOf(txtPassword.getPassword()))) {
181 return true;
182 }
183 } else {
184 //message dialog per file non trovato
185 JOptionPane.showMessageDialog(null, "Errore durante lo splitting delle credenziali", "Eccezione in splitting", JOptionPane.ERROR_MESSAGE);
186 break;
187 }
188 }
189
190 } catch (FileNotFoundException ex) {
191 //message dialog per file non trovato
192 JOptionPane.showMessageDialog(null, "File accessodb.csv non trovato.", "File non trovato", JOptionPane.ERROR_MESSAGE);
193
194 } catch (IOException ex) {
195 //message dialog per lettura errata
196 JOptionPane.showMessageDialog(null, "Eccezione invocata durante la lettura dal file.", "Eccezione in lettura", JOptionPane.ERROR_MESSAGE);
197
198 }
199 //reset dei campi di accesso
200 this.txtPassword.setText("");
201 this.txtUsername.setText("");
202 return false;
203 }
