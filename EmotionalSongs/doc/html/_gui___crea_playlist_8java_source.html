342 public void RegistraPlaylist() {
343 // TODO add your handling code here:
344 if (this.jLstAnteprima.getModel().getSize() == 0) {
345 JOptionPane.showMessageDialog(this, "Aggiungere almeno una canzone.", "Errore", JOptionPane.ERROR_MESSAGE);
346 } else {
347 try {
348 is.createPlaylist(nomePlaylist, playlist, userId);
349 JOptionPane.showMessageDialog(this, "Playlist creata correttamente.", "Info", JOptionPane.INFORMATION_MESSAGE);
350 cleanAll();
351 } catch(ChiaveDuplicata | DatiNonValidi | MyServerException | RemoteException ex){
352 JOptionPane.showMessageDialog(null, "Dati errati", "Errore", JOptionPane.WARNING_MESSAGE);
353 }
354 cleanAll();
355
356 }
