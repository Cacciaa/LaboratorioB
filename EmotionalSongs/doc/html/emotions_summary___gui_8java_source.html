143 private Vector<String> transform(int index, ArrayList<EmozioniCanzone> brani) {

174 private boolean setMedieTabella(JTable tabella) {
175 MediaEmozioni media=null;
176 int avrg = -1;
177 try {
178 //model delle colonne della tabella
179 TableColumnModel df = (TableColumnModel) tabella.getColumnModel();
180 media = this.emo_canzone.getMedia();
181 //for per l'aggiornamento delle colonne della tabella con le medie di ogni emozioni
182 for (int i = 1; i < df.getColumnCount(); i++) {
183 //ottengo la media di ogni emozione specificata all'indice indicato
184 switch(i){
185 case 1 : avrg = this.emo_canzone.getMedia().getAvg_amazement();
186 break;
187 case 2 : avrg = this.emo_canzone.getMedia().getAvg_solemnity();
188 break;
189 case 3 : avrg = this.emo_canzone.getMedia().getAvg_tenderness();
190 break;
191 case 4 : avrg = this.emo_canzone.getMedia().getAvg_nostalgia();
192 break;
193 case 5 : avrg = this.emo_canzone.getMedia().getAvg_calmness();
194 break;
195 case 6 : avrg = this.emo_canzone.getMedia().getAvg_power();
196 break;
197 case 7 : avrg = this.emo_canzone.getMedia().getAvg_joy();
198 break;
199 case 8 : avrg = this.emo_canzone.getMedia().getAvg_tension();
200 break;
201 case 9 : avrg = this.emo_canzone.getMedia().getAvg_sadness();
202 break;
203 }
204 //nel caso in cui ottengo -1, errore durante l'invocazione del get delle medie
205 if (avrg == -1) {
206 return false;
207 } else {
208 //setting della colonna i-esima col valore della media ottenuta
209 df.getColumn(i).setHeaderValue(tabella.getColumnName(i) + "(" + avrg + ")");
210 }
211
212 }
213 return true;
214 //catch dell'eccezione in fase di popolamento delle colonne della tabella
215 } catch (Exception e) {
216 return false;
217 }
